35inline void ImpulseDynamics::linearizeImpulseDynamics(
36 Robot& robot, const ImpulseStatus& impulse_status,
37 const ImpulseSplitSolution& s, ImpulseSplitKKTResidual& kkt_residual) {
38 computeImpulseDynamicsResidual(robot, impulse_status, s);
39 robot.RNEAImpulseDerivatives(s.q, s.dv, data_.dImDdq(), data_.dImDddv);
40 robot.computeImpulseVelocityDerivatives(impulse_status, data_.dCdq(),
41 data_.dCdv());
42 // augment inverse impulse dynamics constraint
43 kkt_residual.lq().noalias() += data_.dImDdq().transpose() * s.beta;
44 kkt_residual.ldv.noalias() += data_.dImDddv.transpose() * s.beta;
45 kkt_residual.lf().noalias() -= data_.dCdv() * s.beta;
46 // augment impulse velocity constraint
47 kkt_residual.lq().noalias() += data_.dCdq().transpose() * s.mu_stack();
48 kkt_residual.lv().noalias() += data_.dCdv().transpose() * s.mu_stack();
49 kkt_residual.ldv.noalias() += data_.dCdv().transpose() * s.mu_stack();
50}

53inline void ImpulseDynamics::condenseImpulseDynamics(
54 Robot& robot, const ImpulseStatus& impulse_status,
55 ImpulseSplitKKTMatrix& kkt_matrix, ImpulseSplitKKTResidual& kkt_residual) {
56 robot.computeMJtJinv(data_.dImDddv, data_.dCdv(), data_.MJtJinv());
57 const int dimv = robot.dimv();
58 const int dimf = impulse_status.dimf();
59 data_.MJtJinv_dImDCdqv().leftCols(dimv).noalias()
60 = data_.MJtJinv() * data_.dImDCdq();
61 data_.MJtJinv_dImDCdqv().topRightCorner(dimv, dimv).noalias()
62 = data_.MJtJinv().topRightCorner(dimv, dimf) * data_.dCdv();
63 data_.MJtJinv_dImDCdqv().bottomRightCorner(dimf, dimv).noalias()
64 = data_.MJtJinv().bottomRightCorner(dimf, dimf) * data_.dCdv();
65 data_.MJtJinv_ImDC().noalias() = data_.MJtJinv() * data_.ImDC();
66
67 data_.Qdvfqv().topRows(dimv).noalias()
68 = (- kkt_matrix.Qdvdv.diagonal()).asDiagonal()
69 * data_.MJtJinv_dImDCdqv().topRows(dimv);
70 data_.Qdvfqv().bottomRows(dimf).noalias()
71 = - kkt_matrix.Qff() * data_.MJtJinv_dImDCdqv().bottomRows(dimf);
72 data_.Qdvfqv().bottomLeftCorner(dimf, dimv).noalias()
73 -= kkt_matrix.Qqf().transpose();
74 data_.ldv() = kkt_residual.ldv;
75 data_.lf() = - kkt_residual.lf();
76 data_.ldv().noalias()
77 -= kkt_matrix.Qdvdv.diagonal().asDiagonal()
78 * data_.MJtJinv_ImDC().head(dimv);
79 data_.lf().noalias() -= kkt_matrix.Qff() * data_.MJtJinv_ImDC().tail(dimf);
80
81 kkt_matrix.Qxx.noalias()
82 -= data_.MJtJinv_dImDCdqv().transpose() * data_.Qdvfqv();
83 kkt_matrix.Qxx.topRows(dimv).noalias()
84 += kkt_matrix.Qqf() * data_.MJtJinv_dImDCdqv().bottomRows(dimf);
85 kkt_residual.lx.noalias()
86 -= data_.MJtJinv_dImDCdqv().transpose() * data_.ldvf();
87 kkt_residual.lq().noalias()
88 += kkt_matrix.Qqf() * data_.MJtJinv_ImDC().tail(dimf);
89
90 kkt_matrix.Fvq() = - data_.MJtJinv_dImDCdqv().topLeftCorner(dimv, dimv);
91 kkt_matrix.Fvv() = Eigen::MatrixXd::Identity(dimv, dimv)
92 - data_.MJtJinv_dImDCdqv().topRightCorner(dimv, dimv);
93 kkt_residual.Fv().noalias() -= data_.MJtJinv_ImDC().head(dimv);
94}
