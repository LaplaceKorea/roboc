62inline void UnconstrDynamics::linearizeUnconstrDynamics(
63 Robot& robot, const double dt, const SplitSolution& s,
64 SplitKKTResidual& kkt_residual) {
65 assert(dt > 0);
66 computeUnconstrDynamicsResidual(robot, s);
67 // augment inverse dynamics constraint
68 robot.RNEADerivatives(s.q, s.v, s.a, dID_dq_, dID_dv_, dID_da_);
69 kkt_residual.lq().noalias() += dt * dID_dq_.transpose() * s.beta;
70 kkt_residual.lv().noalias() += dt * dID_dv_.transpose() * s.beta;
71 kkt_residual.la.noalias() += dt * dID_da_.transpose() * s.beta;
72 kkt_residual.lu.noalias() -= dt * s.beta;
73}

76inline void UnconstrDynamics::condenseUnconstrDynamics(
77 SplitKKTMatrix& kkt_matrix, SplitKKTResidual& kkt_residual) {
78 // condense KKT residual
79 lu_condensed_ = kkt_residual.lu;
80 lu_condensed_.noalias() += kkt_matrix.Quu.diagonal().asDiagonal() * ID_;
81 kkt_residual.lq().noalias() += dID_dq_.transpose() * lu_condensed_;
82 kkt_residual.lv().noalias() += dID_dv_.transpose() * lu_condensed_;
83 kkt_residual.la.noalias() += dID_da_.transpose() * lu_condensed_;
84 // condense KKT Hessian
85 Quu_dID_dq_.noalias() = kkt_matrix.Quu.diagonal().asDiagonal() * dID_dq_;
86 Quu_dID_dv_.noalias() = kkt_matrix.Quu.diagonal().asDiagonal() * dID_dv_;
87 Quu_dID_da_.noalias() = kkt_matrix.Quu.diagonal().asDiagonal() * dID_da_;
88 kkt_matrix.Qqq().noalias() += dID_dq_.transpose() * Quu_dID_dq_;
89 kkt_matrix.Qqv().noalias() += dID_dq_.transpose() * Quu_dID_dv_;
90 kkt_matrix.Qvq() = kkt_matrix.Qqv().transpose();
91 kkt_matrix.Qvv().noalias() += dID_dv_.transpose() * Quu_dID_dv_;
92 kkt_matrix.Qaa.noalias() += dID_da_.transpose() * Quu_dID_da_;
93 // this is actually Qqa and Qqv
94 kkt_matrix.Qqu().noalias() = Quu_dID_dq_.transpose() * dID_da_;
95 kkt_matrix.Qvu().noalias() = Quu_dID_dv_.transpose() * dID_da_;
96}
