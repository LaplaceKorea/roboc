128 template <typename VectorType1, typename VectorType2>
129 static void frictionConeResidual(const double mu,
130 const Eigen::MatrixBase<VectorType1>& f,
131 const Eigen::MatrixBase<VectorType2>& res) {
132 assert(mu > 0);
133 assert(f.size() == 3);
134 assert(res.size() == 5);
135 const_cast<Eigen::MatrixBase<VectorType2>&>(res).coeffRef(0) = - f.coeff(2);
136 const_cast<Eigen::MatrixBase<VectorType2>&>(res).coeffRef(1)
137 = f.coeff(0) - mu * f.coeff(2) / std::sqrt(2);
138 const_cast<Eigen::MatrixBase<VectorType2>&>(res).coeffRef(2)
139 = - f.coeff(0) - mu * f.coeff(2) / std::sqrt(2);
140 const_cast<Eigen::MatrixBase<VectorType2>&>(res).coeffRef(3)
141 = f.coeff(1) - mu * f.coeff(2) / std::sqrt(2);
142 const_cast<Eigen::MatrixBase<VectorType2>&>(res).coeffRef(4)
143 = - f.coeff(1) - mu * f.coeff(2) / std::sqrt(2);
144 }
