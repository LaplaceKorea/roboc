51inline void ImpulseSplitOCP::evalOCP(Robot& robot,
52 const ImpulseStatus& impulse_status,
53 const double t,
54 const ImpulseSplitSolution& s,
55 const Eigen::VectorXd& q_next,
56 const Eigen::VectorXd& v_next,
57 ImpulseSplitKKTResidual& kkt_residual) {
58 assert(q_next.size() == robot.dimq());
59 assert(v_next.size() == robot.dimv());
60 kkt_residual.setImpulseStatus(impulse_status);
61 kkt_residual.setZero();
62 robot.updateKinematics(s.q, s.v+s.dv);
63 stage_cost_ = cost_->computeImpulseCost(robot, cost_data_, t, s);
64 constraints_->evalConstraint(robot, constraints_data_, s);
65 stage_cost_ += constraints_data_.logBarrier();
66 state_equation_.computeStateEquationResidual(robot, s, q_next, v_next,
67 kkt_residual);
68 impulse_dynamics_.computeImpulseDynamicsResidual(robot, impulse_status, s);
69}

72inline void ImpulseSplitOCP::computeKKTResidual(
73 Robot& robot, const ImpulseStatus& impulse_status, const double t,
74 const Eigen::VectorXd& q_prev, const ImpulseSplitSolution& s,

79 kkt_matrix.setImpulseStatus(impulse_status);

94inline void ImpulseSplitOCP::computeKKTSystem(
95 Robot& robot, const ImpulseStatus& impulse_status, const double t,
96 const Eigen::VectorXd& q_prev, const ImpulseSplitSolution& s,
97 const SplitSolution& s_next, ImpulseSplitKKTMatrix& kkt_matrix,
98 ImpulseSplitKKTResidual& kkt_residual) {
99 assert(q_prev.size() == robot.dimq());
100 robot.updateKinematics(s.q, s.v+s.dv);
101 kkt_matrix.setImpulseStatus(impulse_status);
102 kkt_residual.setImpulseStatus(impulse_status);
103 kkt_matrix.setZero();
104 kkt_residual.setZero();
105 stage_cost_ = cost_->quadratizeImpulseCost(robot, cost_data_, t, s,
106 kkt_residual, kkt_matrix);
107 constraints_->condenseSlackAndDual(robot, constraints_data_, s,
108 kkt_matrix, kkt_residual);
109 stage_cost_ += constraints_data_.logBarrier();
110 state_equation_.linearizeStateEquationAlongLieGroup(robot, q_prev, s, s_next,
111 kkt_matrix, kkt_residual);
112 impulse_dynamics_.linearizeImpulseDynamics(robot, impulse_status, s,
113 kkt_residual);
114 impulse_dynamics_.condenseImpulseDynamics(robot, impulse_status,
115 kkt_matrix, kkt_residual);
116}
