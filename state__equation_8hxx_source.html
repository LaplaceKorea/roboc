44template <typename ConfigVectorType, typename TangentVectorType>
45inline void StateEquation::computeStateEquationResidual(
46 const Robot& robot, const double dt, const SplitSolution& s,
47 const Eigen::MatrixBase<ConfigVectorType>& q_next,
48 const Eigen::MatrixBase<TangentVectorType>& v_next,
49 SplitKKTResidual& kkt_residual) {
50 assert(dt > 0);
51 assert(q_next.size() == robot.dimq());
52 assert(v_next.size() == robot.dimv());
53 robot.subtractConfiguration(s.q, q_next, kkt_residual.Fq());

59template <typename ConfigVectorType, typename SplitSolutionType>
60inline void StateEquation::linearizeStateEquation(
61 const Robot& robot, const double dt,
62 const Eigen::MatrixBase<ConfigVectorType>& q_prev, const SplitSolution& s,
63 const SplitSolutionType& s_next, SplitKKTMatrix& kkt_matrix,
64 SplitKKTResidual& kkt_residual) {
65 assert(dt > 0);
66 assert(q_prev.size() == robot.dimq());
67 computeStateEquationResidual(robot, dt, s, s_next.q, s_next.v, kkt_residual);
68 if (robot.hasFloatingBase()) {
69 robot.dSubtractConfiguration_dqf(s.q, s_next.q, kkt_matrix.Fqq());
70 robot.dSubtractConfiguration_dq0(q_prev, s.q, kkt_matrix.Fqq_prev);
71 kkt_residual.lq().template head<6>().noalias()
72 += kkt_matrix.Fqq().template topLeftCorner<6, 6>().transpose()
73 * s_next.lmd.template head<6>();
74 kkt_residual.lq().template head<6>().noalias()
75 += kkt_matrix.Fqq_prev.template topLeftCorner<6, 6>().transpose()
76 * s.lmd.template head<6>();
77 kkt_residual.lq().tail(robot.dimv()-6).noalias()
78 += s_next.lmd.tail(robot.dimv()-6) - s.lmd.tail(robot.dimv()-6);
79 }
80 else {
81 kkt_matrix.Fqq().diagonal().fill(1.);
82 kkt_residual.lq().noalias() += s_next.lmd - s.lmd;
83 }
84 kkt_matrix.Fqv().diagonal().fill(dt);
85 kkt_residual.lv().noalias() += dt * s_next.lmd + s_next.gmm - s.gmm;
86 kkt_residual.la.noalias() += dt * s_next.gmm;
87}
