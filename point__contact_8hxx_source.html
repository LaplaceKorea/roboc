108template <typename VectorType1, typename VectorType2>
109inline void PointContact::computeBaumgarteResidual(
110 const pinocchio::Model& model, const pinocchio::Data& data,
111 const Eigen::MatrixBase<VectorType1>& contact_point,
112 const Eigen::MatrixBase<VectorType2>& baumgarte_residual) const {
113 assert(contact_point.size() == 3);
114 assert(baumgarte_residual.size() == 3);
115 const_cast<Eigen::MatrixBase<VectorType2>&> (baumgarte_residual).noalias()
116 = pinocchio::getFrameClassicalAcceleration(model, data, contact_frame_id_,
117 pinocchio::LOCAL).linear();
118 (const_cast<Eigen::MatrixBase<VectorType2>&> (baumgarte_residual)).noalias()
119 += baumgarte_weight_on_velocity_
120 * pinocchio::getFrameVelocity(model, data, contact_frame_id_,
121 pinocchio::LOCAL).linear();
122 (const_cast<Eigen::MatrixBase<VectorType2>&> (baumgarte_residual)).noalias()
123 += baumgarte_weight_on_position_
124 * (data.oMf[contact_frame_id_].translation()-contact_point);
125}

128template <typename MatrixType1, typename MatrixType2, typename MatrixType3>
129inline void PointContact::computeBaumgarteDerivatives(
130 const pinocchio::Model& model, pinocchio::Data& data,
131 const Eigen::MatrixBase<MatrixType1>& baumgarte_partial_dq,
132 const Eigen::MatrixBase<MatrixType2>& baumgarte_partial_dv,
133 const Eigen::MatrixBase<MatrixType3>& baumgarte_partial_da) {
134 assert(baumgarte_partial_dq.cols() == dimv_);
135 assert(baumgarte_partial_dv.cols() == dimv_);
136 assert(baumgarte_partial_da.cols() == dimv_);
137 assert(baumgarte_partial_dq.rows() == 3);
138 assert(baumgarte_partial_dv.rows() == 3);
139 assert(baumgarte_partial_da.rows() == 3);
140 pinocchio::getFrameAccelerationDerivatives(model, data, contact_frame_id_,
141 pinocchio::LOCAL,
142 frame_v_partial_dq_,
143 frame_a_partial_dq_,
144 frame_a_partial_dv_,
145 frame_a_partial_da_);
146 // Skew matrices and LOCAL frame Jacobian are needed to convert the
147 // frame acceleration derivatives into the "classical" acceleration
148 // derivatives.
149 pinocchio::getFrameJacobian(model, data, contact_frame_id_,
150 pinocchio::LOCAL, J_frame_);
151 v_frame_ = pinocchio::getFrameVelocity(model, data, contact_frame_id_,
152 pinocchio::LOCAL);
153 pinocchio::skew(v_frame_.linear(), v_linear_skew_);
154 pinocchio::skew(v_frame_.angular(), v_angular_skew_);
155 const_cast<Eigen::MatrixBase<MatrixType1>&> (baumgarte_partial_dq)
156 = frame_a_partial_dq_.template topRows<3>();
157 const_cast<Eigen::MatrixBase<MatrixType1>&> (baumgarte_partial_dq).noalias()
158 += v_angular_skew_ * frame_v_partial_dq_.template topRows<3>();
159 const_cast<Eigen::MatrixBase<MatrixType1>&> (baumgarte_partial_dq).noalias()
160 -= v_linear_skew_ * frame_v_partial_dq_.template bottomRows<3>();
161 const_cast<Eigen::MatrixBase<MatrixType2>&> (baumgarte_partial_dv)
162 = frame_a_partial_dv_.template topRows<3>();
163 const_cast<Eigen::MatrixBase<MatrixType2>&> (baumgarte_partial_dv).noalias()
164 += v_angular_skew_ * J_frame_.template topRows<3>();
165 const_cast<Eigen::MatrixBase<MatrixType2>&> (baumgarte_partial_dv).noalias()
166 -= v_linear_skew_ * J_frame_.template bottomRows<3>();
167 const_cast<Eigen::MatrixBase<MatrixType3>&> (baumgarte_partial_da)
168 = frame_a_partial_da_.template topRows<3>();
169 (const_cast<Eigen::MatrixBase<MatrixType1>&> (baumgarte_partial_dq)).noalias()
170 += baumgarte_weight_on_velocity_
171 * frame_v_partial_dq_.template topRows<3>();
172 (const_cast<Eigen::MatrixBase<MatrixType2>&> (baumgarte_partial_dv)).noalias()
173 += baumgarte_weight_on_velocity_
174 * frame_a_partial_da_.template topRows<3>();
175 (const_cast<Eigen::MatrixBase<MatrixType1>&> (baumgarte_partial_dq)).noalias()
176 += baumgarte_weight_on_position_ * data.oMf[contact_frame_id_].rotation()
177 * J_frame_.template topRows<3>();
178}
