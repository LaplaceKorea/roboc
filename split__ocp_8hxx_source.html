56inline void SplitOCP::evalOCP(Robot& robot, const ContactStatus& contact_status,
57 const double t, const double dt,
58 const SplitSolution& s,
59 const Eigen::VectorXd& q_next,
60 const Eigen::VectorXd& v_next,
61 SplitKKTResidual& kkt_residual) {
62 assert(dt > 0);
63 assert(q_next.size() == robot.dimq());
64 assert(v_next.size() == robot.dimv());
65 robot.updateKinematics(s.q, s.v, s.a);
66 kkt_residual.setContactStatus(contact_status);
67 kkt_residual.setZero();
68 stage_cost_ = cost_->computeStageCost(robot, cost_data_, t, dt, s);
69 constraints_->evalConstraint(robot, constraints_data_, s);
70 stage_cost_ += dt * constraints_data_.logBarrier();
71 state_equation_.computeStateEquationResidual(robot, dt, s, q_next, v_next,
72 kkt_residual);
73 contact_dynamics_.computeContactDynamicsResidual(robot, contact_status, s);
74}

263inline double SplitOCP::KKTError(const SplitKKTResidual& kkt_residual,
264 const double dt) const {
265 assert(dt > 0);
266 double err = 0;
267 err += kkt_residual.KKTError();
268 err += (dt*dt) * contact_dynamics_.KKTError();
269 err += (dt*dt) * constraints_data_.KKTError();
270 return err;
271}
