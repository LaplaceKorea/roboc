23inline void Constraints::push_back(
24 const ConstraintComponentBasePtr& constraint) {
25 if (constraint->kinematicsLevel() == KinematicsLevel::PositionLevel) {
26 position_level_constraints_.push_back(constraint);
27 }
28 else if (constraint->kinematicsLevel() == KinematicsLevel::VelocityLevel) {
29 velocity_level_constraints_.push_back(constraint);
30 }
31 else if (constraint->kinematicsLevel() == KinematicsLevel::AccelerationLevel) {
32 acceleration_level_constraints_.push_back(constraint);
