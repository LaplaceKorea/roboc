11inline SplitUnconstrOCP::SplitUnconstrOCP(
12 const Robot& robot, const std::shared_ptr<CostFunction>& cost,
13 const std::shared_ptr<Constraints>& constraints)
14 : cost_(cost),
15 cost_data_(cost->createCostFunctionData(robot)),
16 constraints_(constraints),
17 constraints_data_(constraints->createConstraintsData(robot, 0)),
18 unconstr_dynamics_(robot),
19 use_kinematics_(false),
20 stage_cost_(0) {
21 if (cost_->useKinematics() || constraints_->useKinematics()) {
22 use_kinematics_ = true;
23 }
24 try {
25 if (robot.hasFloatingBase()) {
26 throw std::logic_error(
27 "robot has floating base: robot should have no constraints!");
28 }
29 if (robot.maxPointContacts() > 0) {
30 throw std::logic_error(
31 "robot can have contacts: robot should have no constraints!");
32 }
33 }
34 catch(const std::exception& e) {
35 std::cerr << e.what() << '\n';
36 std::exit(EXIT_FAILURE);
37 }
38}

69inline void SplitUnconstrOCP::evalOCP(Robot& robot, const double t,
70 const double dt, const SplitSolution& s,
71 const Eigen::VectorXd& q_next,
72 const Eigen::VectorXd& v_next,
73 SplitKKTResidual& kkt_residual) {
74 assert(dt > 0);
75 assert(q_next.size() == robot.dimq());
76 assert(v_next.size() == robot.dimv());
77 if (use_kinematics_) {
78 robot.updateKinematics(s.q);
79 }
80 kkt_residual.setZero();
81 stage_cost_ = cost_->computeStageCost(robot, cost_data_, t, dt, s);
82 constraints_->evalConstraint(robot, constraints_data_, s);
83 stage_cost_ += dt * constraints_data_.logBarrier();
84 unconstr::stateequation::computeForwardEulerResidual(dt, s, q_next, v_next,
85 kkt_residual);
86 unconstr_dynamics_.computeUnconstrDynamicsResidual(robot, s);
87}

112inline void SplitUnconstrOCP::computeKKTSystem(Robot& robot, const double t,
113 const double dt,
114 const SplitSolution& s,
115 const SplitSolution& s_next,
116 SplitKKTMatrix& kkt_matrix,
117 SplitKKTResidual& kkt_residual) {
118 assert(dt > 0);
119 if (use_kinematics_) {
120 robot.updateKinematics(s.q);
121 }
122 kkt_matrix.setZero();
123 kkt_residual.setZero();
124 stage_cost_ = cost_->quadratizeStageCost(robot, cost_data_, t, dt, s,
125 kkt_residual, kkt_matrix);
126 constraints_->condenseSlackAndDual(robot, constraints_data_, dt, s,
127 kkt_matrix, kkt_residual);
128 stage_cost_ += dt * constraints_data_.logBarrier();
129 unconstr::stateequation::linearizeForwardEuler(dt, s, s_next,
130 kkt_matrix, kkt_residual);
131 unconstr_dynamics_.linearizeUnconstrDynamics(robot, dt, s, kkt_residual);
132 unconstr_dynamics_.condenseUnconstrDynamics(kkt_matrix, kkt_residual);
133}

174inline double SplitUnconstrOCP::KKTError(const SplitKKTResidual& kkt_residual,
175 const double dt) const {
176 assert(dt > 0);
177 double err = 0;
178 err += kkt_residual.KKTError();
179 err += (dt*dt) * unconstr_dynamics_.KKTError();
180 err += (dt*dt) * constraints_data_.KKTError();
181 return err;
182}
