130 if (lx.size() != 2*dimv_) return false;
131 if (la.size() != dimv_) return false;

137inline bool SplitKKTResidual::isApprox(const SplitKKTResidual& other) const {
138 assert(isDimensionConsistent());
139 assert(other.isDimensionConsistent());
140 if (!Fx.isApprox(other.Fx)) return false;
141 if (!lx.isApprox(other.lx)) return false;
142 if (!la.isApprox(other.la)) return false;
143 if (!lu.isApprox(other.lu)) return false;
144 if (dimf_ > 0) {
145 if (!lf().isApprox(other.lf())) return false;
146 }
147 return true;
