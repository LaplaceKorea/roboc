116inline void DirectMultipleShooting::runParallel(
117 OCP& ocp, aligned_vector<Robot>& robots,
118 const ContactSequence& contact_sequence,
119 const Eigen::VectorXd& q, const Eigen::VectorXd& v, const Solution& s,
120 KKTMatrix& kkt_matrix, KKTResidual& kkt_residual) const {
121 assert(robots.size() == nthreads_);
122 assert(q.size() == robots[0].dimq());
123 assert(v.size() == robots[0].dimv());
124 const int N = ocp.discrete().N();
125 const int N_impulse = ocp.discrete().N_impulse();
126 const int N_lift = ocp.discrete().N_lift();
127 const int N_all = N + 1 + 2*N_impulse + N_lift;
128 #pragma omp parallel for num_threads(nthreads_)
129 for (int i=0; i<N_all; ++i) {
130 if (i < N) {
131 if (ocp.discrete().isTimeStageBeforeImpulse(i)) {
132 assert(!ocp.discrete().isTimeStageBeforeImpulse(i+1));
133 Algorithm::run(
134 ocp[i], robots[omp_get_thread_num()],
135 contact_sequence.contactStatus(ocp.discrete().contactPhase(i)),
136 ocp.discrete().t(i), ocp.discrete().dt(i), q_prev(ocp, q, s, i),
137 s[i], s.impulse[ocp.discrete().impulseIndexAfterTimeStage(i)],
138 kkt_matrix[i], kkt_residual[i]);
139 }
140 else if (ocp.discrete().isTimeStageBeforeLift(i)) {
141 assert(!ocp.discrete().isTimeStageBeforeImpulse(i+1));
142 Algorithm::run(
143 ocp[i], robots[omp_get_thread_num()],
144 contact_sequence.contactStatus(ocp.discrete().contactPhase(i)),
145 ocp.discrete().t(i), ocp.discrete().dt(i), q_prev(ocp, q, s, i),
146 s[i], s.lift[ocp.discrete().liftIndexAfterTimeStage(i)],
147 kkt_matrix[i], kkt_residual[i]);
148 }
149 else if (ocp.discrete().isTimeStageBeforeImpulse(i+1)) {
150 const int impulse_index
151 = ocp.discrete().impulseIndexAfterTimeStage(i+1);
152 Algorithm::run(
153 ocp[i], robots[omp_get_thread_num()],
154 contact_sequence.contactStatus(ocp.discrete().contactPhase(i)),
155 ocp.discrete().t(i), ocp.discrete().dt(i), q_prev(ocp, q, s, i),
156 s[i], s[i+1], kkt_matrix[i], kkt_residual[i],
157 contact_sequence.impulseStatus(impulse_index),
158 ocp.discrete().dt(i+1), kkt_matrix.switching[impulse_index],
159 kkt_residual.switching[impulse_index]);
160 }
161 else {
162 Algorithm::run(
163 ocp[i], robots[omp_get_thread_num()],
164 contact_sequence.contactStatus(ocp.discrete().contactPhase(i)),
165 ocp.discrete().t(i), ocp.discrete().dt(i), q_prev(ocp, q, s, i),
166 s[i], s[i+1], kkt_matrix[i], kkt_residual[i]);
167 }
168 }
169 else if (i == N) {
170 Algorithm::run(ocp.terminal, robots[omp_get_thread_num()],
171 ocp.discrete().t(N), q_prev(ocp, q, s, N), s[N],
172 kkt_matrix[N], kkt_residual[N]);
173 }
174 else if (i < N+1+N_impulse) {
175 const int impulse_index = i - (N+1);
176 const int time_stage_before_impulse
177 = ocp.discrete().timeStageBeforeImpulse(impulse_index);
178 Algorithm::run(ocp.impulse[impulse_index], robots[omp_get_thread_num()],
179 contact_sequence.impulseStatus(impulse_index),
180 ocp.discrete().t_impulse(impulse_index),
181 s[time_stage_before_impulse].q, s.impulse[impulse_index],
182 s.aux[impulse_index], kkt_matrix.impulse[impulse_index],
183 kkt_residual.impulse[impulse_index]);
184 }
185 else if (i < N+1+2*N_impulse) {
186 const int impulse_index = i - (N+1+N_impulse);
187 const int time_stage_after_impulse
188 = ocp.discrete().timeStageAfterImpulse(impulse_index);
189 Algorithm::run(
190 ocp.aux[impulse_index], robots[omp_get_thread_num()],
191 contact_sequence.contactStatus(
192 ocp.discrete().contactPhaseAfterImpulse(impulse_index)),
193 ocp.discrete().t_impulse(impulse_index),
194 ocp.discrete().dt_aux(impulse_index), s.impulse[impulse_index].q,
195 s.aux[impulse_index], s[time_stage_after_impulse],
196 kkt_matrix.aux[impulse_index], kkt_residual.aux[impulse_index]);
197 }
198 else {
199 const int lift_index = i - (N+1+2*N_impulse);
200 const int time_stage_after_lift
201 = ocp.discrete().timeStageAfterLift(lift_index);
202 Algorithm::run(
203 ocp.lift[lift_index], robots[omp_get_thread_num()],
204 contact_sequence.contactStatus(
205 ocp.discrete().contactPhaseAfterLift(lift_index)),
206 ocp.discrete().t_lift(lift_index),
207 ocp.discrete().dt_lift(lift_index), s[time_stage_after_lift-1].q,
208 s.lift[lift_index], s[time_stage_after_lift],
209 kkt_matrix.lift[lift_index], kkt_residual.lift[lift_index]);
210 }
211 }

215inline const Eigen::VectorXd& DirectMultipleShooting::q_prev(const OCP& ocp,
216 const Eigen::VectorXd& q,
217 const Solution& s,
218 const int time_stage) {
219 assert(time_stage >= 0);
220 assert(time_stage <= ocp.discrete().N());
221 if (time_stage == 0) {
222 return q;
223 }
224 else if (ocp.discrete().isTimeStageBeforeImpulse(time_stage-1)) {
225 return s.aux[ocp.discrete().impulseIndexAfterTimeStage(time_stage-1)].q;
226 }
227 else if (ocp.discrete().isTimeStageBeforeLift(time_stage-1)) {
228 return s.lift[ocp.discrete().liftIndexAfterTimeStage(time_stage-1)].q;
229 }
230 else {
231 return s[time_stage-1].q;
232 }
233}
