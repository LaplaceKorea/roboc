30inline void UnconstrBackwardRiccatiRecursionFactorizer::factorizeKKTMatrix(
31 const SplitRiccatiFactorization& riccati_next, const double dt,
32 SplitKKTMatrix& kkt_matrix, SplitKKTResidual& kkt_residual) {
33 assert(dt > 0);
34 // Factorize F
35 kkt_matrix.Qxx.noalias() += riccati_next.P;
36 kkt_matrix.Qxx.bottomRows(dimv_).noalias()
37 += dt * riccati_next.P.topRows(dimv_);
38 kkt_matrix.Qxx.rightCols(dimv_).noalias()
39 += dt * riccati_next.P.leftCols(dimv_);
40 kkt_matrix.Qxx.bottomRightCorner(dimv_, dimv_).noalias()
41 += (dt*dt) * riccati_next.P.topLeftCorner(dimv_, dimv_);
42 // Factorize H
43 kkt_matrix.Qxu.noalias() += dt * riccati_next.P.rightCols(dimv_);
44 kkt_matrix.Qvu().noalias() += (dt*dt) * riccati_next.Pqv();
45 // Factorize G
46 kkt_matrix.Qaa.noalias() += (dt*dt) * riccati_next.Pvv();
47 // Factorize vector term
48 kkt_residual.la.noalias() += dt * riccati_next.P.bottomRows(dimv_)
49 * kkt_residual.Fx;
50 kkt_residual.la.noalias() -= dt * riccati_next.sv();
51}

54inline void UnconstrBackwardRiccatiRecursionFactorizer::
55factorizeRiccatiFactorization(const SplitRiccatiFactorization& riccati_next,
56 SplitKKTMatrix& kkt_matrix,
57 const SplitKKTResidual& kkt_residual,
58 const LQRPolicy& lqr_policy, const double dt,
59 SplitRiccatiFactorization& riccati) {
60 assert(dt > 0);
61 GK_.noalias() = kkt_matrix.Qaa * lqr_policy.K;
62 kkt_matrix.Qxx.noalias() -= lqr_policy.K.transpose() * GK_;
63 // Riccati factorization matrix with preserving the symmetry
64 riccati.P = 0.5 * (kkt_matrix.Qxx + kkt_matrix.Qxx.transpose());
65 // Riccati factorization vector
66 riccati.s = riccati_next.s;
67 riccati.sv().noalias() += dt * riccati_next.sq();
68 riccati.s.noalias() -= riccati_next.P * kkt_residual.Fx;
69 riccati.sv().noalias() -= dt * riccati_next.P.topRows(dimv_)
70 * kkt_residual.Fx;
71 riccati.s.noalias() -= kkt_residual.lx;
72 riccati.s.noalias() -= kkt_matrix.Qxu * lqr_policy.k;
73}
