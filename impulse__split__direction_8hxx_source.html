188inline bool ImpulseSplitDirection::isApprox(
189 const ImpulseSplitDirection& other) const {
190 assert(isDimensionConsistent());
191 assert(other.isDimensionConsistent());
192 assert(dimi()==other.dimi());
193 if (!dx.isApprox(other.dx)) return false;
194 if (!ddvf().isApprox(other.ddvf())) return false;
195 if (!dlmdgmm.isApprox(other.dlmdgmm)) return false;
196 if (!dbetamu().isApprox(other.dbetamu())) return false;
197 return true;
198}
