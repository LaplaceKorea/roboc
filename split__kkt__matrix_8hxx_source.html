199inline bool SplitKKTMatrix::isDimensionConsistent() const {
200 if (Fxx.rows() != 2*dimv_) return false;
201 if (Fxx.cols() != 2*dimv_) return false;
202 if (Fvu.rows() != dimv_) return false;
203 if (Fvu.cols() != dimu_) return false;
204 if (Qxx.rows() != 2*dimv_) return false;
205 if (Qxx.cols() != 2*dimv_) return false;
206 if (Qaa.rows() != dimv_) return false;
207 if (Qaa.cols() != dimv_) return false;
208 if (Qxu.rows() != 2*dimv_) return false;
209 if (Qxu.cols() != dimu_) return false;
210 if (Quu.rows() != dimu_) return false;
211 if (Quu.cols() != dimu_) return false;
212 if (has_floating_base_) {
213 if (Fqq_prev.rows() != dimv_) return false;
214 if (Fqq_prev.cols() != dimv_) return false;
215 }
216 return true;
217}

248inline void SplitKKTMatrix::setRandom() {
249 Fxx.setRandom();
250 Fvu.setRandom();
251 const Eigen::MatrixXd Qxxuu_seed = Eigen::MatrixXd::Random(dimx_+dimu_, dimx_+dimu_);
252 const Eigen::MatrixXd Qxxuu = Qxxuu_seed * Qxxuu_seed.transpose();
253 Qxx = Qxxuu.topLeftCorner(dimx_, dimx_);
254 Qxu = Qxxuu.topRightCorner(dimx_, dimu_);
255 Quu = Qxxuu.bottomRightCorner(dimu_, dimu_);
256 const Eigen::MatrixXd Qaaff_seed = Eigen::MatrixXd::Random(dimv_+dimf_, dimv_+dimf_);
257 const Eigen::MatrixXd Qaaff = Qaaff_seed * Qaaff_seed.transpose();
258 Qaa = Qaaff.topLeftCorner(dimv_, dimv_);
259 Qff() = Qaaff.bottomRightCorner(dimf_, dimf_);
260 Qqf().setRandom();
261 Fqq_prev.setRandom();
262}
