34inline void RiccatiFactorizer::backwardRiccatiRecursion(

36 SplitKKTMatrix& kkt_matrix, SplitKKTResidual& kkt_residual,
37 SplitRiccatiFactorization& riccati, LQRPolicy& lqr_policy) {
38 backward_recursion_.factorizeKKTMatrix(riccati_next, kkt_matrix,
39 kkt_residual);

52inline void RiccatiFactorizer::backwardRiccatiRecursion(
53 const SplitRiccatiFactorization& riccati_next,
54 SplitKKTMatrix& kkt_matrix, SplitKKTResidual& kkt_residual,
55 const SplitSwitchingConstraintJacobian& sc_jacobian,
56 const SplitSwitchingConstraintResidual& sc_residual,
57 SplitRiccatiFactorization& riccati,
58 SplitConstrainedRiccatiFactorization& c_riccati, LQRPolicy& lqr_policy) {
59 backward_recursion_.factorizeKKTMatrix(riccati_next, kkt_matrix, kkt_residual);
60 // Schur complement
61 llt_.compute(kkt_matrix.Quu);
62 assert(llt_.info() == Eigen::Success);
63 c_riccati.setImpulseStatus(sc_jacobian.dimi());
64 c_riccati.Ginv.noalias() = llt_.solve(Eigen::MatrixXd::Identity(dimu_, dimu_));
65 c_riccati.DGinv().transpose().noalias() = llt_.solve(sc_jacobian.Phiu().transpose());
66 c_riccati.S().noalias() = c_riccati.DGinv() * sc_jacobian.Phiu().transpose();
67 llt_s_.compute(c_riccati.S());
68 assert(llt_s_.info() == Eigen::Success);
69 c_riccati.SinvDGinv().noalias() = llt_s_.solve(c_riccati.DGinv());
70 c_riccati.Ginv.noalias() -= c_riccati.SinvDGinv().transpose() * c_riccati.DGinv();
71 lqr_policy.K.noalias() = - c_riccati.Ginv * kkt_matrix.Qxu.transpose();
72 lqr_policy.K.noalias() -= c_riccati.SinvDGinv().transpose() * sc_jacobian.Phix();
73 lqr_policy.k.noalias() = - c_riccati.Ginv * kkt_residual.lu;
74 lqr_policy.k.noalias() -= c_riccati.SinvDGinv().transpose() * sc_residual.P();
75 c_riccati.M().noalias() = llt_s_.solve(sc_jacobian.Phix());
76 c_riccati.M().noalias() -= c_riccati.SinvDGinv() * kkt_matrix.Qxu.transpose();
77 c_riccati.m().noalias() = llt_s_.solve(sc_residual.P());
78 c_riccati.m().noalias() -= c_riccati.SinvDGinv() * kkt_residual.lu;
79 assert(!lqr_policy.K.hasNaN());
80 assert(!lqr_policy.k.hasNaN());
81 assert(!c_riccati.M().hasNaN());
82 assert(!c_riccati.m().hasNaN());
83 backward_recursion_.factorizeRiccatiFactorization(riccati_next, kkt_matrix,
84 kkt_residual, lqr_policy,
85 riccati);
86 c_riccati.DtM.noalias() = sc_jacobian.Phiu().transpose() * c_riccati.M();
87 c_riccati.KtDtM.noalias() = lqr_policy.K.transpose() * c_riccati.DtM;
88 riccati.P.noalias() -= c_riccati.KtDtM;
89 riccati.P.noalias() -= c_riccati.KtDtM.transpose();
90 riccati.s.noalias() -= sc_jacobian.Phix().transpose() * c_riccati.m();
91}

104template <typename SplitDirectionType>
105inline void RiccatiFactorizer::forwardRiccatiRecursion(
106 const SplitKKTMatrix& kkt_matrix, const SplitKKTResidual& kkt_residual,
107 const LQRPolicy& lqr_policy, SplitDirection& d,
108 SplitDirectionType& d_next) const {
109 d.du.noalias() = lqr_policy.K * d.dx;
110 d.du.noalias() += lqr_policy.k;
111 d_next.dx = kkt_residual.Fx;
112 d_next.dx.noalias() += kkt_matrix.Fxx * d.dx;
113 d_next.dv().noalias() += kkt_matrix.Fvu * d.du;
114}
