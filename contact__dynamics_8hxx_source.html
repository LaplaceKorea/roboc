34inline void ContactDynamics::computeContactDynamicsResidual(
35 Robot& robot, const ContactStatus& contact_status, const SplitSolution& s) {
36 data_.setContactStatus(contact_status);
37 has_active_contacts_ = contact_status.hasActiveContacts();
38 robot.setContactForces(contact_status, s.f);
39 robot.RNEA(s.q, s.v, s.a, data_.ID_full());
40 data_.ID().noalias() -= s.u;
41 robot.computeBaumgarteResidual(contact_status, contact_status.contactPoints(),
42 data_.C());
43}

46inline void ContactDynamics::linearizeContactDynamics(
47 Robot& robot, const ContactStatus& contact_status, const double dt,
48 const SplitSolution& s, SplitKKTResidual& kkt_residual) {
49 assert(dt > 0);
50 computeContactDynamicsResidual(robot, contact_status, s);
51 robot.RNEADerivatives(s.q, s.v, s.a, data_.dIDdq(), data_.dIDdv(), data_.dIDda);
52 robot.computeBaumgarteDerivatives(contact_status, data_.dCdq(), data_.dCdv(),
53 data_.dCda());
54 // augment inverse dynamics constraint
55 kkt_residual.lq().noalias() += dt * data_.dIDdq().transpose() * s.beta;
56 kkt_residual.lv().noalias() += dt * data_.dIDdv().transpose() * s.beta;
57 kkt_residual.la.noalias() += dt * data_.dIDda.transpose() * s.beta;
58 if (has_active_contacts_) {
59 kkt_residual.lf().noalias() -= dt * data_.dCda() * s.beta;
60 }
61 if (has_floating_base_) {
62 // augment floating base constraint
63 data_.lu_passive = dt * s.nu_passive;
64 data_.lu_passive.noalias() -= dt * s.beta.template head<kDimFloatingBase>();
65 kkt_residual.lu.noalias() -= dt * s.beta.tail(robot.dimu());
66 }
67 else {
68 kkt_residual.lu.noalias() -= dt * s.beta;
69 }
70 // augment acceleration-level contact constraint
71 if (has_active_contacts_) {
72 kkt_residual.lq().noalias()
73 += dt * data_.dCdq().transpose() * s.mu_stack();
74 kkt_residual.lv().noalias()
75 += dt * data_.dCdv().transpose() * s.mu_stack();
76 kkt_residual.la.noalias() += dt * data_.dCda().transpose() * s.mu_stack();
77 }
78}

81inline void ContactDynamics::condenseContactDynamics(
82 Robot& robot, const ContactStatus& contact_status, const double dt,
83 SplitKKTMatrix& kkt_matrix, SplitKKTResidual& kkt_residual) {
84 assert(dt > 0);
85 const int dimv = robot.dimv();
86 const int dimu = robot.dimu();
87 const int dim_passive = robot.dim_passive();
88 const int dimf = contact_status.dimf();
89 robot.computeMJtJinv(data_.dIDda, data_.dCda(), data_.MJtJinv());
90 data_.MJtJinv_dIDCdqv().noalias() = data_.MJtJinv() * data_.dIDCdqv();
91 data_.MJtJinv_IDC().noalias() = data_.MJtJinv() * data_.IDC();
92
93 data_.Qafqv().topRows(dimv).noalias()
94 = (- kkt_matrix.Qaa.diagonal()).asDiagonal()
95 * data_.MJtJinv_dIDCdqv().topRows(dimv);
96 data_.Qafqv().bottomRows(dimf).noalias()
97 = - kkt_matrix.Qff() * data_.MJtJinv_dIDCdqv().bottomRows(dimf);
98 data_.Qafqv().bottomLeftCorner(dimf, dimv).noalias()
99 -= kkt_matrix.Qqf().transpose();
100 data_.Qafu_full().topRows(dimv).noalias()
101 = kkt_matrix.Qaa.diagonal().asDiagonal()
102 * data_.MJtJinv().topLeftCorner(dimv, dimv);
103 data_.Qafu_full().bottomRows(dimf).noalias()
104 = kkt_matrix.Qff() * data_.MJtJinv().bottomLeftCorner(dimf, dimv);
105 data_.la() = kkt_residual.la;
106 data_.lf() = - kkt_residual.lf();
107 data_.la().noalias()
108 -= kkt_matrix.Qaa.diagonal().asDiagonal()
109 * data_.MJtJinv_IDC().head(dimv);
110 data_.lf().noalias()
111 -= kkt_matrix.Qff() * data_.MJtJinv_IDC().tail(dimf);
112
113 kkt_matrix.Qxx.noalias()
114 -= data_.MJtJinv_dIDCdqv().transpose() * data_.Qafqv();
115 kkt_matrix.Qxx.topRows(dimv).noalias()
116 += kkt_matrix.Qqf() * data_.MJtJinv_dIDCdqv().bottomRows(dimf);
117 if (has_floating_base_) {
118 data_.Qxu_passive.noalias()
119 = - data_.MJtJinv_dIDCdqv().transpose() * data_.Qafu_full().leftCols(dim_passive);
120 data_.Qxu_passive.topRows(dimv).noalias()
121 -= kkt_matrix.Qqf() * data_.MJtJinv().bottomLeftCorner(dimf, dimv).leftCols(dim_passive);
122 kkt_matrix.Qxu.noalias()
123 -= data_.MJtJinv_dIDCdqv().transpose() * data_.Qafu_full().rightCols(dimu);
124 kkt_matrix.Qxu.topRows(dimv).noalias()
125 -= kkt_matrix.Qqf() * data_.MJtJinv().bottomLeftCorner(dimf, dimv).rightCols(dimu);
126 }
127 else {
128 kkt_matrix.Qxu.noalias()
129 -= data_.MJtJinv_dIDCdqv().transpose() * data_.Qafu_full();
130 kkt_matrix.Qxu.topRows(dimv).noalias()
131 -= kkt_matrix.Qqf() * data_.MJtJinv().bottomLeftCorner(dimf, dimv);
132 }
133 kkt_residual.lx.noalias()
134 -= data_.MJtJinv_dIDCdqv().transpose() * data_.laf();
135 kkt_residual.lq().noalias()
136 += kkt_matrix.Qqf() * data_.MJtJinv_IDC().tail(dimf);
137
138 if (has_floating_base_) {
139 data_.Quu_passive_topRight.noalias()
140 = data_.MJtJinv().topRows(dim_passive) * data_.Qafu_full().rightCols(dimu);
141 kkt_matrix.Quu.noalias()
142 += data_.MJtJinv().middleRows(dim_passive, dimu) * data_.Qafu_full().rightCols(dimu);
143 }
144 else {
145 kkt_matrix.Quu.noalias()
146 += data_.MJtJinv().topRows(dimv) * data_.Qafu_full();
147 }
148 if (has_floating_base_) {
149 data_.lu_passive.noalias()
150 += data_.MJtJinv().template topRows<kDimFloatingBase>() * data_.laf();
151 }
152 kkt_residual.lu.noalias()
153 += data_.MJtJinv().middleRows(dim_passive, dimu) * data_.laf();
154
155 kkt_matrix.Fvq() = - dt * data_.MJtJinv_dIDCdqv().topLeftCorner(dimv, dimv);
156 kkt_matrix.Fvv().noalias()
157 = - dt * data_.MJtJinv_dIDCdqv().topRightCorner(dimv, dimv)
158 + Eigen::MatrixXd::Identity(dimv, dimv);
159 kkt_matrix.Fvu = dt * data_.MJtJinv().block(0, dim_passive, dimv, dimu);
160 kkt_residual.Fv().noalias() -= dt * data_.MJtJinv_IDC().head(dimv);
161}

164inline void ContactDynamics::expandPrimal(SplitDirection& d) const {
165 d.daf().noalias() = - data_.MJtJinv_dIDCdqv() * d.dx;
166 d.daf().noalias()

169 d.df().array() *= -1;

173template <typename SplitDirectionType>
174inline void ContactDynamics::expandDual(const double dt,
175 const SplitDirectionType& d_next,
176 SplitDirection& d) {
177 assert(dt > 0);
178 if (has_floating_base_) {
179 d.dnu_passive = data_.lu_passive;
180 d.dnu_passive.noalias() += data_.Quu_passive_topRight * d.du;
181 d.dnu_passive.noalias() += data_.Qxu_passive.transpose() * d.dx;
182 d.dnu_passive.noalias()
183 += dt * data_.MJtJinv().leftCols(dimv_).template topRows<kDimFloatingBase>()
184 * d_next.dgmm();
185 d.dnu_passive.array() *= - (1/dt);
186 }
187 data_.laf().noalias() += data_.Qafqv() * d.dx;
188 data_.laf().noalias() += data_.Qafu() * d.du;
189 data_.la().noalias() += dt * d_next.dgmm();
190 d.dbetamu().noalias() = - data_.MJtJinv() * data_.laf() * (1/dt);
191}
